1.给你两个绝对路径地址，然后你去把file1中的内容拷贝到file2中。

ofstream：用于文件的写入：

  • 1.创建ofstream对象：

      ofstream outputFile;

  • 2.打开文件并写入内容：

      outputFile.open("example.txt"); // 打开名为example.txt的文件进行写入

  if (outputFile.is_open()) {
      outputFile << "Hello, World!" << std::endl; // 将字符串写入文件
      outputFile << "This is a sample text." << std::endl;
  } else {
      std::cout << "Failed to open the file." << std::endl; // 如果文件打开失败，输出错误信息
  }

  outputFile.close(); // 关闭文件流

ifstream：用于文件的读取：

  • 1.创建ifstream inputFile

  • 2.打开文件并读取内容：

      inputFile.open("example.txt"); // 打开名为example.txt的文件进行读取

  if (inputFile.is_open()) {
      std::string line;
      while (std::getline(inputFile, line)) {
          std::cout << line << std::endl; // 输出文件中的每一行内容
      }
  } else {
      std::cout << "Failed to open the file." << std::endl; // 如果文件无法打开，输出错误信息
  }

  inputFile.close(); // 关闭文件流

getline

  • 1.std::getline(输入流, 目标字符串);
    getline(std::cin, line):从输入中读取一行文本，同时将其存在line。

  • 2.从文件中读取一行

    while (std::getline(file, line)) {
  std::cout << line << std::endl;
}

  • 3.自定义分隔符

      std::getline(stream, word, ',')

代码：

#include <iostream>
#include <fstream>
#include <string>

int main() {
std::string file1Path, file2Path;

std::cout << "请输入第一个文件的绝对路径 (file1): ";
std::cin >> file1Path;

std::cout << "请输入第二个文件的绝对路径 (file2): ";
std::cin >> file2Path;

std::ifstream file1(file1Path);
if (!file1) {
    std::cerr << "无法打开文件: " << file1Path << std::endl;
    return 1;
}

std::ofstream file2(file2Path);
if (!file2) {
    std::cerr << "无法打开文件: " << file2Path << std::endl;
    return 1;
}

std::string line;
while (std::getline(file1, line)) {
    file2 << line << "\n";
}

file1.close();
file2.close();

std::cout << "文件拷贝成功！" << std::endl;

return 0;

}
